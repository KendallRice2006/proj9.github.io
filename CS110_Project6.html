# 1. Aggregate medal data by sport
medals_by_sport = df.groupby('sport')[['gold', 'silver', 'bronze']].sum().reset_index()
medals_by_sport['total_medals'] = medals_by_sport['gold'] + medals_by_sport['silver'] + medals_by_sport['bronze']

# Sort by total medals for better visualization
medals_by_sport_sorted = medals_by_sport.sort_values(by='total_medals', ascending=False)

# Visualization 1: Heatmap for medal counts per sport
plt.figure(figsize=(14, 10))
sns.heatmap(medals_by_sport_sorted.set_index('sport')[['gold', 'silver', 'bronze']],
            annot=True, fmt='d', cmap='YlGnBu', linewidths=.5, cbar_kws={'label': 'Number of Medals'})
plt.title('Medal Counts (Gold, Silver, Bronze) per Sport (Rio 2016 Olympics)', fontsize=16)
plt.xlabel('Medal Type', fontsize=12)
plt.ylabel('Sport', fontsize=12)
plt.tight_layout()
plt.show()

# Visualization 2: Stacked bar chart for medal distribution per sport (as before)
# Melt the dataframe to prepare for stacked bar plot
medals_melted = medals_by_sport_sorted.melt(id_vars=['sport'], value_vars=['gold', 'silver', 'bronze'],
                                            var_name='medal_type', value_name='count')

plt.figure(figsize=(14, 8))
sns.barplot(x='count', y='sport', hue='medal_type', data=medals_melted, dodge=False,
            palette={'gold': 'gold', 'silver': 'silver', 'bronze': '#CD7F32'})
plt.title('Distribution of Gold, Silver, and Bronze Medals per Sport (Rio 2016 Olympics)', fontsize=16)
plt.xlabel('Number of Medals', fontsize=12)
plt.ylabel('Sport', fontsize=12)
plt.legend(title='Medal Type', title_fontsize='13', fontsize='11')
plt.grid(axis='x', linestyle='--', alpha=0.7)
plt.tight_layout()
plt.show()
